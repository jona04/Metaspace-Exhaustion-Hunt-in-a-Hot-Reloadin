package io.harbor.textd.cli;

import io.harbor.textd.core.OutputFormat;
import io.harbor.textd.core.ProcessingResult;
import io.harbor.textd.core.TextDaemon;

import java.nio.file.Path;
import java.util.List;

public final class HotReloadStressRunner {
    private static final String SAMPLE = "alpha beta gamma";

    private HotReloadStressRunner() {
    }

    public static void main(String[] args) throws Exception {
        Path pluginJar = Path.of(args[0]);
        int cycles = args.length > 1 ? Integer.parseInt(args[1]) : 500;

        try (TextDaemon daemon = new TextDaemon(pluginJar, 2)) {
            for (int generation = 1; generation <= cycles; generation++) {
                daemon.reload(generation);
                assertOutputs(generation, daemon.processBatch(List.of(SAMPLE, SAMPLE)));
                assertOutput(generation, daemon.process(SAMPLE));
                if (generation % 3 == 0) {
                    assertOutputs(generation, daemon.processBatch(List.of(SAMPLE, SAMPLE)));
                }
            }
        }

        System.out.println("stress-ok cycles=" + cycles);
    }

    private static void assertOutputs(long generation, List<ProcessingResult> results) {
        for (ProcessingResult result : results) {
            assertOutput(generation, result);
        }
    }

    private static void assertOutput(long generation, ProcessingResult result) {
        String expected = OutputFormat.expectedFor(generation, SAMPLE);
        if (!expected.equals(result.output())) {
            throw new IllegalStateException("Unexpected output at generation " + generation + ": " + result.output());
        }
    }
}

package io.harbor.textd.core;

import java.nio.file.Path;
import java.util.Queue;
import java.util.concurrent.ConcurrentLinkedQueue;
import java.util.concurrent.atomic.AtomicReference;

final class GenerationRegistry {
    private final Path pluginJar;
    private final AtomicReference<GenerationLease> active = new AtomicReference<>();
    private final Queue<GenerationLease> retired = new ConcurrentLinkedQueue<>();

    GenerationRegistry(Path pluginJar) {
        this.pluginJar = pluginJar;
    }

    void activate(long generation) throws Exception {
        GenerationLease next = new GenerationLease(LoadedGeneration.load(pluginJar, generation));
        GenerationLease previous = active.getAndSet(next);
        if (previous != null) {
            retired.add(previous);
            previous.retire();
            sweepRetired();
        }
    }

    GenerationLease active() {
        GenerationLease lease = active.get();
        if (lease == null) {
            throw new IllegalStateException("No plugin generation is loaded");
        }
        return lease;
    }

    int pendingRetirements() {
        sweepRetired();
        return retired.size();
    }

    void close() {
        GenerationLease current = active.getAndSet(null);
        if (current != null) {
            retired.add(current);
            current.retire();
        }
        sweepRetired();
    }

    private void sweepRetired() {
        retired.removeIf(GenerationLease::isClosed);
    }
}
