package io.harbor.textd.core;

import java.nio.file.Path;
import java.util.Queue;
import java.util.concurrent.ConcurrentLinkedQueue;
import java.util.concurrent.atomic.AtomicReference;

final class GenerationRegistry {
    private final Path pluginJar;
    private final AtomicReference<GenerationLease> active = new AtomicReference<>();
    private final Queue<GenerationLease> retired = new ConcurrentLinkedQueue<>();

    GenerationRegistry(Path pluginJar) {
        this.pluginJar = pluginJar;
    }

    void activate(long generation) throws Exception {
        GenerationLease next = new GenerationLease(LoadedGeneration.load(pluginJar, generation));
        GenerationLease previous = active.getAndSet(next);
        if (previous != null) {
            retired.add(previous);
            previous.retire();
            sweepRetired();
        }
    }

    GenerationLease active() {
        GenerationLease lease = active.get();
        if (lease == null) {
            throw new IllegalStateException("No plugin generation is loaded");
        }
        return lease;
    }

    int pendingRetirements() {
        sweepRetired();
        return retired.size();
    }

    void close() {
        GenerationLease current = active.getAndSet(null);
        if (current != null) {
            retired.add(current);
            current.retire();
        }
        sweepRetired();
    }

    private void sweepRetired() {
        retired.removeIf(GenerationLease::isClosed);
    }
}

package io.harbor.textd.core;

import io.harbor.textd.api.PluginModule;
import io.harbor.textd.api.PluginModuleFactory;

import java.net.URL;
import java.net.URLClassLoader;
import java.nio.file.Path;

final class LoadedGeneration implements AutoCloseable {
    private static final String FACTORY_CLASS = "io.harbor.textd.customer.CustomerPluginFactory";

    private final long generation;
    private final URLClassLoader classLoader;
    private final PluginModule module;

    private LoadedGeneration(long generation, URLClassLoader classLoader, PluginModule module) {
        this.generation = generation;
        this.classLoader = classLoader;
        this.module = module;
    }

    static LoadedGeneration load(Path pluginJar, long generation) throws Exception {
        URLClassLoader classLoader = new URLClassLoader(new URL[] {pluginJar.toUri().toURL()}, PluginModuleFactory.class.getClassLoader());
        Class<?> factoryType = classLoader.loadClass(FACTORY_CLASS);
        PluginModuleFactory factory = (PluginModuleFactory) factoryType.getDeclaredConstructor().newInstance();
        return new LoadedGeneration(generation, classLoader, factory.open());
    }

    long generation() {
        return generation;
    }

    PluginModule module() {
        return module;
    }

    @Override
    public void close() throws Exception {
        module.close();
        classLoader.close();
    }
}
