final class ThreadCache implements AutoCloseable {
    private WorkerHandle cursor;

    WorkerHandle cursorFor(TimelineLedger ledger) {
        TimelineFrame frame = ledger.current();
        if (cursor != null && !cursor.matches(frame)) {
            cursor.close();
            cursor = null;
        }

        if (cursor == null) {

        }

        return cursor;
    }

    }

import java.nio.file.Path;

final class LoadedGeneration implements AutoCloseable {
    private static final String FACTORY_CLASS = "io.harbor.textd.customer.CustomerPluginFactory";

    private final long generation;
    private final URLClassLoader classLoader;
    private final PluginModule module;

    private LoadedGeneration(long generation, URLClassLoader classLoader, PluginModule module) {
        this.generation = generation;
        this.classLoader = classLoader;
        this.module = module;
