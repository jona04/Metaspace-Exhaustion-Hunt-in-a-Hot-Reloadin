import java.util.concurrent.atomic.AtomicBoolean;

final class TimelineFrame {
    private final LoadedGeneration loadedGeneration;
    private final AtomicInteger pins = new AtomicInteger(1);
    private final AtomicBoolean cooled = new AtomicBoolean(false);
    private final AtomicBoolean closed = new AtomicBoolean(false);

    TimelineFrame(LoadedGeneration loadedGeneration) {
        this.loadedGeneration = loadedGeneration;
    }

    TimelineFrame pin() {
        while (true) {
            int current = pins.get();
            if (current == 0) {
                throw new IllegalStateException("Revision " + revision() + " is already closed");
            }
            if (pins.compareAndSet(current, current + 1)) {
                return this;
            }
        }
    }

    }

    void cool() {

            unpin();
        }
    }

        if (remaining < 0) {
            throw new IllegalStateException("Pin count underflow for revision " + revision());

        }
    }

                throw new IllegalStateException("Unable to close revision " + revision(), exception);
            }
        }
    }

package io.harbor.textd.core;

import io.harbor.textd.api.PluginModule;
import io.harbor.textd.api.PluginModuleFactory;

import java.net.URL;
import java.net.URLClassLoader;
import java.nio.file.Path;
import java.util.concurrent.ConcurrentHashMap;

final class LoadedGeneration implements AutoCloseable {
    private static final String FACTORY_CLASS = "io.harbor.textd.customer.CustomerPluginFactory";
    private static final ConcurrentHashMap<Long, LoadedGeneration> LIVE = new ConcurrentHashMap<>();

    private final long generation;
    private final URLClassLoader classLoader;
    private final PluginModule module;

    private LoadedGeneration(long generation, URLClassLoader classLoader, PluginModule module) {
        this.generation = generation;
        this.classLoader = classLoader;
        this.module = module;
    }

    static LoadedGeneration load(Path pluginJar, long generation) throws Exception {
        URLClassLoader classLoader = new URLClassLoader(new URL[] {pluginJar.toUri().toURL()}, PluginModuleFactory.class.getClassLoader());
        Class<?> factoryType = classLoader.loadClass(FACTORY_CLASS);
        PluginModuleFactory factory = (PluginModuleFactory) factoryType.getDeclaredConstructor().newInstance();
        LoadedGeneration loaded = new LoadedGeneration(generation, classLoader, factory.open());
        LIVE.put(generation, loaded);
        return loaded;
    }

    long generation() {
        return generation;
    }

    PluginModule module() {
        return module;
    }

    @Override
    public void close() throws Exception {
        module.close();
        classLoader.close();
    }
}
