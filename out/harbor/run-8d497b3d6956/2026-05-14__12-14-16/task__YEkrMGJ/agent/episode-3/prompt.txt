import java.nio.file.Path;
import java.util.Queue;
import java.util.concurrent.ConcurrentLinkedQueue;
import java.util.concurrent.atomic.AtomicReference;

final class TimelineLedger {
    private final Path pluginJar;
    private final AtomicReference<TimelineFrame> current = new AtomicReference<>();
    private final Queue<TimelineFrame> shadow = new ConcurrentLinkedQueue<>();

    TimelineLedger(Path pluginJar) {
        this.pluginJar = pluginJar;
    }

    void install(long revision) throws Exception {
        TimelineFrame next = new TimelineFrame(LoadedGeneration.load(pluginJar, revision));
        TimelineFrame previous = current.getAndSet(next);
        if (previous != null) {
            shadow.add(previous);
            previous.cool();
            pruneShadow();
        }
    }

    TimelineFrame current() {
        TimelineFrame frame = current.get();
        if (frame == null) {
            throw new IllegalStateException("No plugin generation is loaded");
        }
        return frame;
    }

    boolean isBalanced() {
        pruneShadow();
        return shadow.isEmpty();
    }

    void close() {
        TimelineFrame frame = current.getAndSet(null);
        if (frame != null) {
            shadow.add(frame);
            frame.cool();
        }
        pruneShadow();
    }

    private void pruneShadow() {
        shadow.removeIf(TimelineFrame::isDrained);
    }
}

package io.harbor.textd.core;

import io.harbor.textd.api.PluginModule;

import java.util.concurrent.atomic.AtomicBoolean;
import java.util.concurrent.atomic.AtomicInteger;

final class TimelineFrame {
    private final LoadedGeneration loadedGeneration;
    private final AtomicInteger pins = new AtomicInteger(1);
    private final AtomicBoolean cooled = new AtomicBoolean(false);
    private final AtomicBoolean closed = new AtomicBoolean(false);

    TimelineFrame(LoadedGeneration loadedGeneration) {
        this.loadedGeneration = loadedGeneration;
    }

    TimelineFrame pin() {
        while (true) {
            int current = pins.get();
            if (current == 0) {
                throw new IllegalStateException("Revision " + revision() + " is already closed");
            }
            if (pins.compareAndSet(current, current + 1)) {
                return this;
            }
        }
    }

    long revision() {
        return loadedGeneration.generation();
    }

    PluginModule module() {
        return loadedGeneration.module();
    }

    void cool() {
        if (cooled.compareAndSet(false, true)) {
            unpin();
        }
    }

    void unpin() {
        int remaining = pins.decrementAndGet();
        if (remaining < 0) {
            throw new IllegalStateException("Pin count underflow for revision " + revision());
        }
        if (remaining == 0 && cooled.get()) {
            closeLoadedGeneration();
        }
    }

    boolean isDrained() {
        return closed.get();
    }

    private void closeLoadedGeneration() {
        if (closed.compareAndSet(false, true)) {
            try {
                loadedGeneration.close();
            } catch (Exception exception) {
                throw new IllegalStateException("Unable to close revision " + revision(), exception);
            }
        }
    }
}

package io.harbor.textd.core;

import io.harbor.textd.api.TextProcessor;

final class WorkerHandle implements AutoCloseable {
    private final TimelineFrame frame;
    private final long revision;
    private final TextProcessor processor;

    WorkerHandle(TimelineFrame frame) {
        this.frame = frame.pin();
        this.revision = frame.revision();
        this.processor = frame.module().processorFor(revision);
    }

    boolean matches(TimelineFrame frame) {
        return revision == frame.revision();
    }

    ProcessingResult process(String record) {
        String output = processor.process(record);
        return new ProcessingResult(revision, output, Thread.currentThread().getName());
    }

    @Override
    public void close() {
        frame.unpin();
    }
}
